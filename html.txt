const TODAY = new Date();
const pad = n => String(n).padStart(2,"0");
const toKey = d => d.getFullYear()+"-"+pad(d.getMonth()+1)+"-"+pad(d.getDate());
const TODAY_KEY = toKey(TODAY);
const shiftKey = i => { const d=new Date(TODAY); d.setDate(TODAY.getDate()-i); return toKey(d); };
const daysInMonth = (y,m) => new Date(y,m+1,0).getDate();
const firstDay    = (y,m) => new Date(y,m,1).getDay();

// ══════════════════════════════════════════════════
// SCORE
// ══════════════════════════════════════════════════
function getScore(dk) {
  const ch = STATE.checks;
  if (ch["rest_"+dk]) return {pct: ch["stretch_"+dk]?105:100, isRest:true};
  const normal = HABITS.filter(h=>h.id!=="satisfied");
  const done   = normal.filter(h=>!!ch[h.id+"_"+dk]).length;
  const bonus  = ch["satisfied_"+dk]?1:0;
  return {pct: Math.min(Math.round(((done+bonus)/normal.length)*100),105), isRest:false};
}

function dotColor(pct, isRest) {
  if (isRest)   return C.blue;
  if (pct>=100) return C.green;
  if (pct>=70)  return C.gold;
  if (pct>=40)  return C.yellow;
  if (pct>0)    return C.red;
  return C.dark;
}

function scColor(p) {
  return p>=100?C.green:p>=70?C.gold:p>=40?C.yellow:C.red;
}

function calcStreak() {
  let s=0; const d=new Date(TODAY);
  while(s<365){ const dk=toKey(d); if(getScore(dk).pct>0){s++;d.setDate(d.getDate()-1);}else break; }
  return s;
}

// ══════════════════════════════════════════════════
// RENDER

function render() {
  const todayPct = getScore(TODAY_KEY).pct;
  const week7    = Array.from({length:7},(_,i)=>getScore(shiftKey(i)).pct);
  const weekAvg  = Math.round(week7.reduce((a,b)=>a+b,0)/7);
  const dm0      = daysInMonth(TODAY.getFullYear(),TODAY.getMonth());
  const monthAvg = Math.round(Array.from({length:dm0},(_,i)=>getScore(toKey(new Date(TODAY.getFullYear(),TODAY.getMonth(),i+1))).pct).reduce((a,b)=>a+b,0)/dm0);
  const streak   = calcStreak();

  // Streak
  const sb = document.getElementById("streak-box");
  if (streak > 0) {
    sb.style.display = "flex";
    document.getElementById("streak-num").textContent = streak+" jour"+(streak>1?"s":"");
  } else { sb.style.display = "none"; }

  // Cards
  const mainVal = currentPeriod==="day"?todayPct:currentPeriod==="week"?weekAvg:monthAvg;
  const secVal  = currentPeriod==="day"?weekAvg:currentPeriod==="week"?monthAvg:todayPct;
  const mainLbl = currentPeriod==="day"?"Aujourd'hui":currentPeriod==="week"?"Moy. semaine":"Moy. mois";
  const secLbl  = currentPeriod==="day"?"Semaine":currentPeriod==="week"?"Mois":"Aujourd'hui";
  document.getElementById("card1-lbl").textContent = mainLbl;
  document.getElementById("card1-val").textContent = mainVal+"%";
  document.getElementById("card1-val").style.color = scColor(mainVal);
  document.getElementById("card2-lbl").textContent = secLbl;
  document.getElementById("card2-val").textContent = secVal+"%";
  document.getElementById("card2-val").style.color = scColor(secVal);

  // Piliers
  document.getElementById("piliers-title").textContent = "Piliers — "+(currentPeriod==="day"?"Aujourd'hui":currentPeriod==="week"?"Semaine":"Mois");
  const barsEl = document.getElementById("piliers-bars");
  barsEl.innerHTML = "";
  Object.keys(CATS).forEach(cat => {
    const hh = HABITS.filter(h=>h.cat===cat);
    let done=0,total=0;
    if (currentPeriod==="day") {
      total=hh.length; done=hh.filter(h=>!!STATE.checks[h.id+"_"+TODAY_KEY]).length;
    } else if (currentPeriod==="week") {
      const days=Array.from({length:7},(_,i)=>shiftKey(i));
      total=hh.length*7; done=hh.reduce((a,h)=>a+days.filter(dk=>!!STATE.checks[h.id+"_"+dk]).length,0);
    } else {
      const y=TODAY.getFullYear(),m=TODAY.getMonth(),dm=daysInMonth(y,m);
      const keys=Array.from({length:dm},(_,i)=>toKey(new Date(y,m,i+1)));
      total=hh.length*dm; done=hh.reduce((a,h)=>a+keys.filter(dk=>!!STATE.checks[h.id+"_"+dk]).length,0);
    }
    const pct = total?Math.round((done/total)*100):0;
    const {label,color} = CATS[cat];
    barsEl.innerHTML += `
      <div class="bar-row">
        <div class="bar-label"><span>${label}</span><span style="color:${color};font-weight:700">${pct}%</span></div>
        <div class="bar-bg"><div class="bar-fill" style="width:${Math.min(pct,100)}%;background:${color};${pct>=100?"box-shadow:0 0 8px "+color+"66":""}"></div></div>

function renderChart(week7) {
  const pts = [...week7].reverse().map((v,i)=>{
    const d=new Date(TODAY); d.setDate(TODAY.getDate()-(6-i));
    return {v, lbl:d.toLocaleDateString("fr-FR",{weekday:"short"}).slice(0,1).toUpperCase()};
  });
  const W=280,H=70,p=8,xs=(W-p*2)/6;
  const coords=pts.map((pt,i)=>({x:p+i*xs, y:H-p-((pt.v/105)*(H-p*2)), ...pt}));
  const line=coords.map((c,i)=>(i===0?"M":"L")+c.x+","+c.y).join(" ");
  const area=line+" L"+coords[coords.length-1].x+","+H+" L"+coords[0].x+","+H+" Z";
  let svg = "";
  [0,50,100].forEach(v=>{
    const y=H-p-((v/105)*(H-p*2));
    svg+=<line x1="${p}" x2="${W-p}" y1="${y}" y2="${y}" stroke="#1e1e1e" stroke-width="1"/>;
  });
  svg+=<path d="${area}" fill="rgba(240,165,0,0.08)"/>;
  svg+=<path d="${line}" fill="none" stroke="#f0a500" stroke-width="2" stroke-linejoin="round" stroke-linecap="round"/>;
  coords.forEach(c=>{
    svg+=<circle cx="${c.x}" cy="${c.y}" r="3.5" fill="${dotColor(c.v,false)}" stroke="#0a0a0a" stroke-width="1.5"/>;
    svg+=<text x="${c.x}" y="${H+8}" text-anchor="middle" font-size="8" fill="#666">${c.lbl}</text>;
  });
  document.getElementById("mini-chart").innerHTML = svg;
}

function renderHabits() {
  const container = document.getElementById("habits-container");
  container.innerHTML = "";
  const isRest = !!STATE.checks["rest_"+TODAY_KEY];

  Object.keys(CATS).forEach(cat => {
    const {label,color} = CATS[cat];
    const habits = HABITS.filter(h=>h.cat===cat);
    const isTraining = cat==="entrainement";

    let html = <div class="cat-title"><div class="cat-bar" style="background:${color}"></div><span class="cat-label" style="color:${color}">${label}</span></div>;

    if (isTraining) {
      const restChecked = !!STATE.checks["rest_"+TODAY_KEY];
      html += habitRowHTML("rest_"+TODAY_KEY, "🛌", "Jour de repos (séance non prévue)", C.blue, restChecked);
      if (restChecked) {
        const stretchChecked = !!STATE.checks["stretch_"+TODAY_KEY];
        html += <div style="padding-left:14px;margin-top:6px">;
        html += habitRowHTML("stretch_"+TODAY_KEY, "🧘", "Étirements / récupération douce (bonus)", "#a78bfa", stretchChecked, true);
        html += </div>;
      }
    }

    if (!isTraining || !isRest) {
      habits.forEach(h => {
        const checked = !!STATE.checks[h.id+"_"+TODAY_KEY];
        html += habitRowHTML(h.id+"_"+TODAY_KEY, h.icon, h.label, color, checked, h.id==="satisfied");
      });
    }

    container.innerHTML += html;
  });

  // Bind clicks
  container.querySelectorAll(".hrow").forEach(el => {
    el.addEventListener("click", () => toggleHabit(el.dataset.key));
  });
}

function habitRowHTML(key, icon, label, color, checked, isBonus=false) {
  const rgb = hexToRgb(color);
  return `<div class="hrow${checked?" checked":""}" data-key="${key}" data-color="${color}" style="--row-color:${color};--row-rgb:${rgb};${checked?"background:"+color+"18;border-color:"+color+"55":""}">
    <div class="hrow-check${checked?" on":""}"><span>${checked?"✓":""}</span></div>
    <span class="hrow-icon">${icon}</span>
    <span class="hrow-text">${label}</span>
    ${isBonus?'<span class="bonus-badge">BONUS</span>':""}
  </div>`;
}

function hexToRgb(hex) {
  const r=parseInt(hex.slice(1,3),16), g=parseInt(hex.slice(3,5),16), b=parseInt(hex.slice(5,7),16);
  return r+","+g+","+b;
}

function renderMonth() {
  const dm   = daysInMonth(calYear,calMonth);
  const fd   = firstDay(calYear,calMonth);
  const off  = fd===0?6:fd-1;
  const name = new Date(calYear,calMonth,1).toLocaleDateString("fr-FR",{month:"long",year:"numeric"});
  document.getElementById("month-name").textContent = name;

  const days = Array.from({length:dm},(_,i)=>{
    const d=new Date(calYear,calMonth,i+1);
    const dk=calYear+"-"+pad(calMonth+1)+"-"+pad(i+1);
    const future=d>TODAY;
    const {pct,isRest}=getScore(dk);
    return{day:i+1,dk,pct,isRest,future,isToday:dk===TODAY_KEY};
  });

  const filled=days.filter(d=>!d.future);
  const avg=filled.length?Math.round(filled.reduce((a,d)=>a+d.pct,0)/filled.length):0;
  document.getElementById("m-perfect").textContent=filled.filter(d=>d.pct>=100).length;
  document.getElementById("m-partial").textContent=filled.filter(d=>d.pct>0&&d.pct<100).length;
  document.getElementById("m-avg").textContent=avg+"%";

  // Headers
  const headers = document.getElementById("cal-headers");
  headers.innerHTML = ["L","M","M","J","V","S","D"].map(d=><div class="cal-day-lbl">${d}</div>).join("");

  // Grid
  const grid = document.getElementById("cal-grid");
  let html = Array.from({length:off}).map(()=>"<div></div>").join("");
  days.forEach(({day,dk,pct,isRest,future,isToday}) => {
    const col = future?"#2a2a2a":dotColor(pct,isRest);
    const hasNote = STATE.notes&&STATE.notes[dk];
    const tap = !future&&pct>0;
    html += `<div class="cal-day${tap?" tappable":""}" 
      style="background:${future?"transparent":col+"22"};border-color:${isToday?"#f0a500":future||pct===0?"#1e1e1e":col}"
      ${tap?'onclick="openModal(\''+dk+'\')"':""}>
      <span class="cal-day-num" style="color:${future?"#2a2a2a":isToday?"#f0a500":"#fff"};font-weight:${isToday?900:500}">${day}</span>
      ${!future&&pct>0?<span class="cal-day-pct" style="color:${col}">${pct}%</span>:""}
      ${isRest&&!future?<span style="font-size:7px">🛌</span>:""}
      ${hasNote&&!future?<div class="cal-dot"></div>:""}
    </div>`;
  });
  grid.innerHTML = html;

  // Legend
  document.getElementById("cal-legend").innerHTML = [
    {c:C.green,l:"100%"},{c:C.gold,l:"70–99%"},{c:C.yellow,l:"40–69%"},{c:C.red,l:"<40%"},{c:C.blue,l:"Repos"}
  ].map(({c,l})=><div class="legend-item"><div class="legend-dot" style="background:${c}"></div>${l}</div>).join("");
}

function changeMonth(dir) {
  calMonth += dir;
  if (calMonth>11){calMonth=0;calYear++;}
  if (calMonth<0){calMonth=11;calYear--;}
  renderMonth();
}

let syncTimeout = null;
function syncToSheet() {
  if (!STATE.athlete || STATE.athlete.trim()==="") return;
  if (!SCRIPT_URL || SCRIPT_URL==="COLLE_TON_URL_ICI") return;
  clearTimeout(syncTimeout);
  showSync("pending","⏳ Synchronisation...");
  syncTimeout = setTimeout(async ()=>{
    try {
      await fetch(SCRIPT_URL, {
        method:"POST",
        mode:"no-cors",
        headers:{"Content-Type":"application/json"},
        body: JSON.stringify({
          athlete: STATE.athlete,
          date: TODAY_KEY,
          checks: STATE.checks,
          note: STATE.notes[TODAY_KEY]||"",
          score: getScore(TODAY_KEY).pct
        })
      });
      showSync("ok","✅ Synchronisé");
    } catch(e) { showSync("err","❌ Erreur de sync"); }
  }, 1500);
}

function showSync(type, msg) {
  const el = document.getElementById("sync-status");
  el.style.display="block"; el.className="sync "+type; el.textContent=msg;
  if (type==="ok") setTimeout(()=>{ el.style.display="none"; },3000);
}
